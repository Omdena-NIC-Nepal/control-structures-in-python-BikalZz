# Task 1: while loop
def while_loop():
    number = 0
    while number <= 20:
        if number % 2 == 0:
            print(number)
        if number == 16:
            break
        number += 1

# Task 2: for loop with continue
def for_loop_continue():
    for num in range(1, 16):
        if num % 3 == 0:
            continue
        print(num)

# Task 3: if-else statement
def number_classification():
    try:
        number = int(input("Enter a number: "))
        if number == 0:
            print("zero")
        elif number > 0:
            print("positive")
        else:
            print("negative")
    except ValueError:
        print("Enter a valid number.")

# Task 4: Nested loops
def multiplication_table():
    for i in range(1, 6):
        for j in range(1, 6):
            print(f'{i} x {j} = {i*j}')
